\def\myframebox#1{%
\setbox0%
\hbox{\vrule
	\vbox{\hrule
		\vskip\vmez
		\hbox{\hskip\hmez #1\hskip\hmez}
		\vskip\vmez
	\hrule
	}%
\vrule
}%
{\everycr{}\tabskip0pt \ooalign{\phantom#1\cr \hss\hbox to0pt{\hss \vbox to0pt{\vss\box0 \vskip-\vmez\vskip-0.4pt}\hss}\hss}}%
}
